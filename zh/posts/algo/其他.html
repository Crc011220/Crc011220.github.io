public String[] longestPalindrome(String inPutStr) {

private boolean isPalindrome(String s) {
    int left = 0;
    int right = s.length() - 1;
    
    while (left < right) {
        if (s.charAt(left) != s.charAt(right)) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}

private long countWays(char[] ch, int start, int end, boolean target) {
    // åŸºç¡€æƒ…å†µï¼šåªæœ‰ä¸€ä¸ªæ•°å­—

    // å°è¯•æ¯ä¸ªè¿ç®—ç¬¦ä½œä¸ºä¸»è¿ç®—ç¬¦

        long rightFalse = countWays(ch, i + 1, end, false);
        
        // æ ¹æ®è¿ç®—ç¬¦è®¡ç®—ç»“æœ
        if (op == '&') {
            if (target) {

                count += leftTrue * rightFalse + leftFalse * rightTrue + leftFalse * rightFalse;
            }
        } else if (op == '|') {
            if (target) {
                count += leftTrue * rightTrue + leftTrue * rightFalse + leftFalse * rightTrue;
            } else {
                count += leftFalse * rightFalse;
            }
        } else if (op == '^') {
            if (target) {
                count += leftTrue * rightFalse + leftFalse * rightTrue;
            } else {
                count += leftTrue * rightTrue + leftFalse * rightFalse;
            }
        }
    }

public class QuickSort {
    
    public static void quickSort(int[] arr, int low, int high) {
        if (low < high) {
            // åˆ†åŒºæ“ä½œï¼Œè¿”å›åŸºå‡†å…ƒç´ ä½ç½®
            int pivot = partition(arr, low, high);
            
            // é€’å½’æ’åºåŸºå‡†å…ƒç´ å·¦è¾¹
            quickSort(arr, low, pivot - 1);
            
            // é€’å½’æ’åºåŸºå‡†å…ƒç´ å³è¾¹
            quickSort(arr, pivot + 1, high);

    }
    
    private static int partition(int[] arr, int low, int high) {
        // é€‰æ‹©æœ€åä¸€ä¸ªå…ƒç´ ä½œä¸ºåŸºå‡†
        int pivot = arr[high];
        int i = low - 1; // å°äºåŸºå‡†å…ƒç´ çš„æŒ‡é’ˆ
        
        for (int j = low; j < high; j++) {
            // å¦‚æœå½“å‰å…ƒç´ å°äºç­‰äºåŸºå‡†
            if (arr[j] <= pivot) {
                i++; // å°†å°äºåŸºå‡†å…ƒç´ çš„æŒ‡é’ˆå³ç§»
                swap(arr, i, j); // å°†å½“å‰å…ƒç´ ä¸å°äºåŸºå‡†å…ƒç´ çš„æŒ‡é’ˆäº¤æ¢
            }
        }
        
        // å°†åŸºå‡†æ”¾åˆ°æ­£ç¡®ä½ç½®
        swap(arr, i + 1, high);
        return i + 1; // è¿”å›åŸºå‡†å…ƒç´ çš„ç´¢å¼•
    }
    
    private static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
    
    // ä¸»æ–¹æ³•
    public static void quickSort(int[] arr) {
        quickSort(arr, 0, arr.length - 1);
    }
}

private static long calculateFactorial(int x) {
        if (x == 0 || x == 1) {
            return 1;
        }
        long factorial = 1;
        for (int i = 2; i <= x; i++) {
            factorial *= i;
        }
        return factorial;
    }
    
private static long calculateFactorial1(int x) {
    if (x == 0 || x == 1) {
        return 1;
    }
    return x * calculateFactorial1(x - 1);
}

public boolean isValidIP(String ip) {
    if (ip == null || ip.length() == 0) {
        return false;
    }
    // åˆ†å‰²IPåœ°å€ï¼ˆæ³¨æ„splitä¼šå¿½ç•¥ç©ºå­—ç¬¦ä¸²ï¼Œéœ€é¢å¤–å¤„ç†è¿ç»­ç‚¹å·çš„æƒ…å†µï¼‰
    String[] parts = ip.split("\\.");
    if (parts.length != 4) {
        return false;
    }
    for (String part : parts) {
        // æ£€æŸ¥æ¯ä¸ªéƒ¨åˆ†çš„é•¿åº¦ï¼ˆ0-3ä½ï¼‰
        if (part.length() == 0 || part.length() > 3) {

        // æ£€æŸ¥æ˜¯å¦æœ‰å‰å¯¼é›¶ï¼ˆé•¿åº¦>1ä¸”ä»¥0å¼€å¤´ï¼‰
        if (part.length() > 1 && part.charAt(0) == '0') {
            return false;
        }
        // æ£€æŸ¥æ˜¯å¦å…¨ä¸ºæ•°å­—
        for (char c : part.toCharArray()) {
            if (!Character.isDigit(c)) {
                return false;
            }
        }
        // æ£€æŸ¥æ•°å€¼æ˜¯å¦åœ¨0-255èŒƒå›´å†…
        int num;
        try {
            num = Integer.parseInt(part);

            return false;
        }
    }
    return true;
