    public void backtrack(List<List<String>> res, List<Integer> path, int n, int row,
                          Set<Integer> cols, Set<Integer> diag1, Set<Integer> diag2) {

            path.add(col); // å¤„ç†èŠ‚ç‚¹
            cols.add(col); // å¤„ç†èŠ‚ç‚¹
            diag1.add(row - col); // å¤„ç†èŠ‚ç‚¹
            diag2.add(row + col); // å¤„ç†èŠ‚ç‚¹
            backtrack(res, path, n, row + 1, cols, diag1, diag2); // é€’å½’
            path.remove(path.size() - 1); // å›æº¯

    private List<String> buildBoard(List<Integer> path, int n) {

        boolean found = dfs(i+1, j, index+1)
